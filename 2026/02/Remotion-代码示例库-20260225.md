## 文本动画示例

### 逐字出现效果

```typescript
export const CharacterAnimation = ({ text, delay = 0 }) => {
  const frame = useCurrentFrame();
  return (
    <div>
      {text.split('').map((char, i) => (
        <span
          key={i}
          style={{
            opacity: frame > i * 2 + delay ? 1 : 0,
            display: 'inline-block'
          }}
        >
          {char}
        </span>
      ))}
    </div>
  );
};
```

### 渐变淡入

```typescript
export const TextFadeIn = ({ text, duration = 30 }) => {
  const frame = useCurrentFrame();
  const opacity = interpolate(
    frame,
    [0, duration],
    [0, 1],
    { extrapolateRight: 'clamp' }
  );

  return <div style={{ opacity }}>{text}</div>;
};
```

## 数据可视化示例

### 柱状图动画

```typescript
export const AnimatedBarChart = ({ data }) => {
  const frame = useCurrentFrame();
  const progress = Math.min(frame / 60, 1);

  return (
    <div style={{ display: 'flex', gap: '10px' }}>
      {data.map((item) => (
        <div
          key={item.name}
          style={{
            height: `${item.value * progress}px`,
            width: '40px',
            backgroundColor: '#FF6B35'
          }}
        />
      ))}
    </div>
  );
};
```

## 音频处理示例

### 音量响应式动画

```typescript
export const AudioReactive = ({ audioSrc }) => {
  const frame = useCurrentFrame();
  const audioData = useAudioData(audioSrc);
  
  if (!audioData) return null;

  const waveData = audioData.channelWaveData[0];
  const volume = waveData[Math.floor(frame * waveData.length / 300)];

  return (
    <div style={{ scale: 1 + volume * 0.5 }} />
  );
};
```
